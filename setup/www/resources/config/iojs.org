server {
    listen *:80;
    server_name iojs.org www.iojs.org;

    return 301 https://nodejs.org;
}

server {
    listen [::]:443 ssl http2;
    listen *:443 ssl http2;
    server_name iojs.org www.iojs.org;

    ssl_certificate ssl/iojs_chained.crt;
    ssl_certificate_key ssl/iojs.key;
    ssl_trusted_certificate ssl/iojs_chained.crt;

    return 301 https://nodejs.org;
}
